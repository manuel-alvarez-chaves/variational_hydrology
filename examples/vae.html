class VAE(nn.Module):
    def __init__(self, num_input, num_hidden):
        super().__init__()
        self.input_size = num_input
        self.hidden_size = num_hidden

        # AutoEncoder
        self.encoder = nn.LSTM(input_size=num_input, hidden_size=num_hidden, batch_first=True)
        self.decoder = nn.Linear(num_hidden, 1)
        self.scale = nn.Sequential(
            nn.Linear(num_hidden, num_hidden),
            nn.Sigmoid(),
        )

        # VAE
        self.mu = nn.Linear(num_hidden, num_hidden)
        self.log_var = nn.Linear(num_hidden, num_hidden)

    def reparametrize(self, mu, log_var):
        std = torch.exp(0.5 * log_var)
        eps = torch.randn_like(std).requires_grad_(False)
        return mu + eps * std
    
    def encode(self, x):
        _, (h_n, _) = self.encoder(x)
        return self.scale(h_n[-1]) # many-to-one
    
    def decode(self, encoded, z):
        return self.decoder(encoded * (1 + z))
    
    def forward(self, x):
        encoded = self.encode(x)
        mu = self.mu(encoded)
        log_var = self.log_var(encoded)
        z = self.reparametrize(mu, log_var)
        decoded = self.decode(encoded, z)
        return encoded, decoded, mu, log_var
    
    def sample(self, x, num_samples):
        num_batches = x.shape[0]
        with torch.no_grad():
            encoded = self.encode(x)
            encoded = encoded.unsqueeze(1).repeat(1, num_samples, 1)
            z = torch.randn(num_batches, num_samples, self.hidden_size)
            out = self.decoder(encoded * (1 + z))
        return out

def loss_function(x_hat, x, mu, log_var):
    return (loss_reconstruction(x_hat, x), loss_regularization(mu, log_var))

def loss_reconstruction(x_hat, x):
    # MSE
    for data in [x_hat, x]:
        lambd = 0.3
        data = ((data + 1)**lambd - 1) / lambd
    loss = nn.functional.mse_loss(x_hat, x)
    return loss

def loss_regularization(mu, log_var):
    # KL Divergence
    mu, log_var = mu, log_var
    return -0.5 * torch.sum(1 + log_var - mu.pow(2) - log_var.exp())

lrs = [1e-4] * 10

# model = VAE(5, 10)
for epoch, lr in enumerate(lrs):
    optimizer = torch.optim.Adam(model.parameters(), lr)

    total_loss_reconstruction = 0.0
    total_loss_regularization = 0.0
    for batch in tqdm(dl_train, ascii=True):
        x_d, _, _, _ = batch.values()
        x_d = x_d[:, :, :-1]
        encoded, decoded, mu, log_var = model(x_d[:, :, :-1])
        loss = loss_function(decoded, x_d[:, -1, [-1]], mu, log_var)
        total_loss_reconstruction += loss[0].item()
        total_loss_regularization += loss[1].item()

        loss = loss[0] + 1e-2 * loss[1]

        # Backward
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
    avg_loss = [loss / len(dl_train) for loss in (total_loss_reconstruction, total_loss_regularization)]

    tqdm.write(f"Epoch {epoch + 1:3d} | Loss: (reconstruction: {avg_loss[0]:.2f}, regularization: {avg_loss[1]:.2f})")
